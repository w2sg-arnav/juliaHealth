Part 4

Description: Build a convolutional network and find the optimal learning rate.

First, I define a sequential chain of layers to be called with their associated activation functions. I based this architecture off of LeNet-5 (Y. Lecun, L. Bottou, Y. Bengio and P. Haffner, "Gradient-based learning applied to document recognition," in Proceedings of the IEEE, vol. 86, no. 11, pp. 2278-2324, Nov. 1998, doi: 10.1109/5.726791.) and referenced the architecture design from the FluxML Model Zoo documentation (https://github.com/FluxML/model-zoo/tree/master/vision/convmnist). The original _LeNet-5 implementation used sigmoid activation functions but it has become best practice in computer vision tasks to use ReLU activation functions instead:

model = Chain(
    Conv((5, 5), 1=>6, relu),
    MaxPool((2, 2)),
    Conv((5, 5), 6=>16, relu),
    MaxPool((2, 2)),
    Flux.flatten,
    Dense(256 => 120, relu),
    Dense(120 => 84, relu), 
    Dense(84 => 10),
    softmax
)

Then I define some helper functions:

#=
    Convenience function to have data altogether in one 
    place for training. Specifically encoded for MNIST
    and to support batch training.
=#
function simple_loader(data::MNIST=train_data; batchsize::Int=64)
    x4dim = reshape(data.features, 28, 28, 1, :)
    yhot = Flux.onehotbatch(data.targets, 0:9)
    Flux.DataLoader((x4dim, yhot); batchsize, shuffle=true)
end

#=
    Calculates accuracy for a given model
=#
function simple_accuracy(model, data::MNIST=test_data)
    (x, y) = only(simple_loader(data; batchsize=length(data)))
    y_hat = model(x)
    iscorrect = Flux.onecold(y_hat) .== Flux.onecold(y)
    acc = round(100 * mean(iscorrect); digits=2)
end

Then I define the stochastic gradient descent optimizer I want to use during training optimization:

learning_rate = 1e-3
optimizer = Descent

simple_optim = Flux.setup(
    optimizer(learning_rate), 
    model, 
);

Batch training data:

train_loader = simple_loader(train_data, batchsize = 256)

Train the model and report loss (cross entropy), training accuracy, and test accuracy:

epochs = 100
global losses = []
global train_accs = []
global test_accs = []

for epoch in 1:epochs
    model_loss = 0.0
    for (x, y) in train_loader
        curr_loss, gradients = Flux.withgradient(m -> Flux.crossentropy(m(x), y), model)
        Flux.update!(simple_optim, model, gradients[1])
        model_loss += curr_loss / length(train_loader)
    end
    push!(train_accs, simple_accuracy(model, train_data))
    push!(test_accs, simple_accuracy(model, test_data))
    push!(losses, model_loss)
    @info "Epoch = $epoch" model_loss
end

Setting Up Convolutional Neural Network

First, I define a sequential chain of layers to be called with their associated activation functions:

model = Flux.Chain(
    Conv((3, 3), 1 => 4, relu, pad = 1),
    MaxPool((2, 2)),
    Conv((3, 3), 4 => 8, relu, pad = 1),
    MaxPool((2, 2)),
    Conv((3, 3),8 => 16, relu, pad = 1),
    MaxPool((2, 2)),
    Conv((3, 3), 16 => 32, relu, pad = 1),
    MaxPool((2, 2)),
    Flux.flatten,
    Dense(floor(Int, 128/16 * 128/16 * 32) => 20, Flux.relu),
    Flux.Dropout(0.1),
    Dense(20 => 4),
    Flux.softmax
)

Then I define some helper functions:

#=
    Convenience function to have data altogether in one 
    place for training. Specifically encoded for MNIST
    and to support batch training.
=#
function simple_loader(data; batchsize::Int=64)
    x4dim = reshape(data.features, 128, 128, 1, :)
    yhot = Flux.onehotbatch(data.targets, 1:4)
    Flux.DataLoader((x4dim, yhot); batchsize, shuffle=true)
end

#=
    Calculates accuracy for a given model
=#
function simple_accuracy(model, data)
    (x, y) = only(simple_loader(data; batchsize=length(data.targets)))
    y_hat = model(x)
    iscorrect = Flux.onecold(y_hat) .== Flux.onecold(y)
    acc = round(100 * mean(iscorrect); digits=2)
end

Then I define an Adam optimizer I want to use during training optimization:

learning_rate = 1e-3
optimizer = Adam

optim = Flux.setup(
    optimizer(learning_rate), 
    model
);

epochs = 100

train_loader = simple_loader(training; batchsize = 256)
for epoch in 1:epochs
    model_loss = 0.0
    for (x, y) in train_loader
        curr_loss, gradients = Flux.withgradient(m -> Flux.crossentropy(m(x), y), model)
        model_loss += curr_loss / length(train_loader)
        Flux.update!(optim, model, gradients[1])
    end
    @info "After epoch = $epoch" model_loss
end
